CREATE TABLE User (
    id INT AUTO_INCREMENT PRIMARY KEY,
    firstName VARCHAR(255),
    lastName VARCHAR(255),
    phoneNumber VARCHAR(20),
    email VARCHAR(255) UNIQUE
);

CREATE TABLE Dentist (
    userId INT,
    specialization VARCHAR(255),
    FOREIGN KEY (userId) REFERENCES User(id)
);

CREATE TABLE Address (
    id INT AUTO_INCREMENT PRIMARY KEY,
    city VARCHAR(255),
    street VARCHAR(255),
    state VARCHAR(255)
);

CREATE TABLE Patient (
    userId INT,
    dateOfBirth DATE,
    mailingAddressId INT,
    FOREIGN KEY (userId) REFERENCES User(id),
    FOREIGN KEY (mailingAddressId) REFERENCES Address(id)
);

CREATE TABLE Surgery (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255),
    telNumber VARCHAR(20),
    addressId INT,
    FOREIGN KEY (addressId) REFERENCES Address(id)
);

CREATE TABLE Appointment (
    id INT AUTO_INCREMENT PRIMARY KEY,
    dentistId INT,
    patientId INT,
    date DATE,
    time TIME,
    surgeryId INT,
    FOREIGN KEY (dentistId) REFERENCES Dentist(userId),
    FOREIGN KEY (patientId) REFERENCES Patient(userId),
    FOREIGN KEY (surgeryId) REFERENCES Surgery(id)
);
